\lecture{7}{}{Estimation -- The Kalman Filter}

\begin{intuition}
	The Kalman filter is the estimation counterpart of LQR. LQR chooses a control law that minimizes a quadratic cost for a linear system; the Kalman filter chooses an estimator that minimizes mean-square estimation error for a linear stochastic system. The same Riccati structure reappears, but with the roles of dynamics, actuation, and sensing transposed. This lecture follows the planned path: algorithm, gain derivation, covariance recursion, steady-state behavior, duality, innovations, EKF, and a full scalar worked example.
\end{intuition}

\subsection{Problem Setup}

\begin{definition}[Linear stochastic state-space model]
	The discrete-time Kalman-filter setting is the linear stochastic system
	\begin{align*}
		x_{k+1} &= A x_k + B u_k + w_k, \\
		y_k &= C x_k + v_k,
	\end{align*}
	where
	\[
		x_k \in \mathbb{R}^n,
		\qquad
		u_k \in \mathbb{R}^m,
		\qquad
		y_k \in \mathbb{R}^p.
	\]
	The random variables satisfy
	\[
		\mathbb{E}[w_k]=0,
		\qquad
		\mathbb{E}[v_k]=0,
	\]
	with covariances
	\[
		\mathbb{E}[w_k w_k^\top]=Q_w \succeq 0,
		\qquad
		\mathbb{E}[v_k v_k^\top]=R_v \succ 0.
	\]
\end{definition}

\begin{notation}[Noise assumptions]
	The standard Kalman assumptions are:
	\begin{itemize}
		\item the initial state error, process noise, and measurement noise are mutually independent,
		\item the sequences $\{w_k\}$ and $\{v_k\}$ are white,
		\item often, though not always, they are assumed Gaussian.
	\end{itemize}
	Gaussianity upgrades the Kalman filter from a best linear estimator to the exact minimum-variance conditional-mean estimator.
\end{notation}

\begin{definition}[State estimation problem]
	Given the measurement history
	\[
		\mathcal{Y}_k = \sigma(y_0,\dots,y_k),
	\]
	the state-estimation problem is to construct estimates of $x_k$ from available data. Two standard estimates are:
	\begin{itemize}
		\item the \emph{prediction}
		      \[
			      \hat x_{k|k-1} = \mathbb{E}[x_k \mid \mathcal{Y}_{k-1}],
		      \]
		\item the \emph{filtered estimate}
		      \[
			      \hat x_{k|k} = \mathbb{E}[x_k \mid \mathcal{Y}_{k}].
		      \]
	\end{itemize}
\end{definition}

\begin{definition}[Prediction and filtering error covariances]
	The corresponding error covariances are
	\[
		P_{k|k-1}
		=
		\mathbb{E}\left[(x_k-\hat x_{k|k-1})(x_k-\hat x_{k|k-1})^\top\right],
	\]
	\[
		P_{k|k}
		=
		\mathbb{E}\left[(x_k-\hat x_{k|k})(x_k-\hat x_{k|k})^\top\right].
	\]
\end{definition}

\begin{theorem}[Optimality criterion]
	Let $\mathcal{G}$ be any sigma-algebra and let $x$ be square integrable. Then the conditional expectation
	\[
		\mathbb{E}[x \mid \mathcal{G}]
	\]
	is the unique minimizer of
	\[
		\mathbb{E}\bigl[\|x-\xi\|^2\bigr]
	\]
	over all square-integrable $\mathcal{G}$-measurable random vectors $\xi$.
	Consequently, under the linear-Gaussian state-space model, the Kalman filter is the minimum-mean-square-error estimator.
\end{theorem}
\begin{proof}
	Fix any $\mathcal{G}$-measurable $\xi$ and write
	\[
		x-\xi
		=
		\bigl(x-\mathbb{E}[x \mid \mathcal{G}]\bigr)
		+
		\bigl(\mathbb{E}[x \mid \mathcal{G}] - \xi\bigr).
	\]
	Squaring and taking expectations gives
	\begin{align*}
		\mathbb{E}\bigl[\|x-\xi\|^2\bigr]
		&=
		\mathbb{E}\bigl[\|x-\mathbb{E}[x \mid \mathcal{G}]\|^2\bigr]
		+
		\mathbb{E}\bigl[\|\mathbb{E}[x \mid \mathcal{G}] - \xi\|^2\bigr] \\
		&\qquad
		+
		2\mathbb{E}\Bigl[
		\bigl(x-\mathbb{E}[x \mid \mathcal{G}]\bigr)^\top
		\bigl(\mathbb{E}[x \mid \mathcal{G}] - \xi\bigr)
		\Bigr].
	\end{align*}
	The cross term vanishes by the defining orthogonality property of conditional expectation: the error
	\[
		x-\mathbb{E}[x \mid \mathcal{G}]
	\]
	is orthogonal to every square-integrable $\mathcal{G}$-measurable random vector. Hence
	\[
		\mathbb{E}\bigl[\|x-\xi\|^2\bigr]
		=
		\mathbb{E}\bigl[\|x-\mathbb{E}[x \mid \mathcal{G}]\|^2\bigr]
		+
		\mathbb{E}\bigl[\|\mathbb{E}[x \mid \mathcal{G}] - \xi\|^2\bigr].
	\]
	The second term is nonnegative and vanishes only when
	\[
		\xi=\mathbb{E}[x \mid \mathcal{G}]
	\]
	almost surely. This proves optimality and uniqueness.
\end{proof}

\subsection{The Kalman Filter Algorithm}

\begin{definition}[Prediction step]
	Given $\hat x_{k-1|k-1}$ and $P_{k-1|k-1}$, the Kalman predictor is
	\[
		\hat x_{k|k-1}=A\hat x_{k-1|k-1}+Bu_{k-1},
	\]
	\[
		P_{k|k-1}=A P_{k-1|k-1}A^\top + Q_w.
	\]
\end{definition}

\begin{definition}[Correction step]
	After receiving measurement $y_k$, the Kalman correction is
	\[
		L_k = P_{k|k-1}C^\top\left(CP_{k|k-1}C^\top + R_v\right)^{-1},
	\]
	\[
		\hat x_{k|k}=\hat x_{k|k-1}+L_k\left(y_k-C\hat x_{k|k-1}\right),
	\]
	\[
		P_{k|k}=(I-L_k C)P_{k|k-1}.
	\]
	The matrix $L_k$ is the \emph{Kalman gain}.
\end{definition}

\begin{remark}[Predict, then correct]
	The filter alternates between two distinct pieces of information:
	\begin{itemize}
		\item the model, which predicts where the state should move,
		\item the measurements, which reveal where the state appears to be.
	\end{itemize}
	The Kalman gain determines how much trust is assigned to each.
\end{remark}

\begin{algorithm}[H]
	\DontPrintSemicolon
	\caption{Discrete-Time Kalman Filter}
	\KwIn{$A$, $B$, $C$, process covariance $Q_w$, measurement covariance $R_v$, initial estimate $\hat x_{0|-1}$, initial covariance $P_{0|-1}$}
	\For{$k=0,1,2,\dots$}{
		Receive input $u_{k-1}$ if applicable and measurement $y_k$\;
		Compute the Kalman gain
		\[
			L_k \gets P_{k|k-1}C^\top\left(CP_{k|k-1}C^\top + R_v\right)^{-1}
		\]\;
		Update the state estimate
		\[
			\hat x_{k|k} \gets \hat x_{k|k-1}+L_k(y_k-C\hat x_{k|k-1})
		\]\;
		Update the covariance
		\[
			P_{k|k} \gets (I-L_k C)P_{k|k-1}
		\]\;
		Predict one step ahead
		\[
			\hat x_{k+1|k} \gets A\hat x_{k|k}+Bu_k
		\]
		and
		\[
			P_{k+1|k} \gets A P_{k|k}A^\top + Q_w
		\]\;
	}
\end{algorithm}

\subsection{Derivation of the Kalman Gain}

\begin{theorem}[Kalman gain from covariance minimization]
	Fix a predicted estimate $\hat x_{k|k-1}$ with prediction covariance
	\[
		P^- := P_{k|k-1}.
	\]
	Among all affine updates of the form
	\[
		\hat x_{k|k}=\hat x_{k|k-1}+L_k(y_k-C\hat x_{k|k-1}),
	\]
	the choice that minimizes
	\[
		\operatorname{tr}(P_{k|k})
	\]
	is
	\[
		L_k = P^- C^\top(CP^- C^\top + R_v)^{-1}.
	\]
\end{theorem}
\begin{proof}
	Let
	\[
		e^- = x_k-\hat x_{k|k-1}.
	\]
	Since
	\[
		y_k = Cx_k + v_k,
	\]
	the updated estimation error is
	\begin{align*}
		e^+
		&= x_k-\hat x_{k|k} \\
		&= x_k-\hat x_{k|k-1}-L_k(y_k-C\hat x_{k|k-1}) \\
		&= (I-L_k C)e^- - L_k v_k.
	\end{align*}
	Using independence of $e^-$ and $v_k$, we obtain
	\[
		P_{k|k}
		=
		(I-L_k C)P^-(I-L_k C)^\top + L_k R_v L_k^\top.
	\]
	Expanding the trace gives
	\begin{align*}
		\operatorname{tr}(P_{k|k})
		&=
		\operatorname{tr}(P^-)
		-2\operatorname{tr}(L_k C P^-)
		+
		\operatorname{tr}\left(L_k(CP^-C^\top+R_v)L_k^\top\right).
	\end{align*}
	Set
	\[
		S_k = CP^-C^\top+R_v.
	\]
	Because $R_v \succ 0$, we have $S_k \succ 0$. Using standard matrix differentiation,
	\[
		\nabla_{L_k}\operatorname{tr}(P_{k|k})
		=
		-2P^-C^\top + 2L_k S_k.
	\]
	Setting the gradient equal to zero yields
	\[
		L_k S_k = P^- C^\top,
	\]
	hence
	\[
		L_k = P^- C^\top S_k^{-1}
		=
		P^- C^\top(CP^-C^\top+R_v)^{-1}.
	\]
	Since the objective is strictly convex in $L_k$ because $S_k \succ 0$, this minimizer is unique.
\end{proof}

\begin{remark}[Joseph stabilized covariance form]
	The covariance update can also be written as
	\[
		P_{k|k}
		=
		(I-L_k C)P_{k|k-1}(I-L_k C)^\top + L_k R_v L_k^\top.
	\]
	This is called the \emph{Joseph form}. Algebraically it is equivalent to
	\[
		P_{k|k}=(I-L_k C)P_{k|k-1}
	\]
	when $L_k$ is the optimal Kalman gain, but numerically the Joseph form preserves symmetry and positive semidefiniteness more robustly in finite precision.
\end{remark}

\subsection{Error Covariance and Riccati Recursion}

\begin{theorem}[Riccati recursion for the error covariance]
	The Kalman-filter covariance recursion is
	\[
		P_{k|k-1}=A P_{k-1|k-1}A^\top + Q_w,
	\]
	\[
		P_{k|k}
		=
		P_{k|k-1}
		-
		P_{k|k-1}C^\top\left(CP_{k|k-1}C^\top+R_v\right)^{-1}CP_{k|k-1}.
	\]
	Combining the two steps gives the one-step prior recursion
	\[
		P_{k+1|k}
		=
		A P_{k|k-1}A^\top + Q_w
		-
		A P_{k|k-1}C^\top\left(CP_{k|k-1}C^\top+R_v\right)^{-1}CP_{k|k-1}A^\top.
	\]
\end{theorem}
\begin{proof}
	The prediction covariance formula follows from
	\[
		x_k-\hat x_{k|k-1} = A(x_{k-1}-\hat x_{k-1|k-1}) + w_{k-1},
	\]
	combined with independence of the previous estimation error and the process noise.

	The filtering covariance formula was derived in the proof of the Kalman gain theorem:
	\[
		P_{k|k}
		=
		(I-L_k C)P_{k|k-1}(I-L_k C)^\top + L_k R_v L_k^\top.
	\]
	Substituting the optimal gain
	\[
		L_k = P_{k|k-1}C^\top\left(CP_{k|k-1}C^\top+R_v\right)^{-1}
	\]
	and simplifying yields
	\[
		P_{k|k}
		=
		P_{k|k-1}
		-
		P_{k|k-1}C^\top\left(CP_{k|k-1}C^\top+R_v\right)^{-1}CP_{k|k-1}.
	\]
	Applying the prediction step once more gives the combined recursion.
\end{proof}

\begin{remark}[Riccati structure]
	The covariance recursion is a Riccati equation in disguise. Structurally it mirrors the LQR Riccati equation, but with sensing matrices and noise covariances replacing actuation matrices and control penalties.
\end{remark}

\begin{proposition}[Limiting behavior of the Kalman gain with respect to noise levels]
	Let
	\[
		L_k = P_{k|k-1}C^\top\left(CP_{k|k-1}C^\top+R_v\right)^{-1}.
	\]
	Then:
	\begin{itemize}
		\item as the measurement noise covariance $R_v$ decreases, the gain increases and the filter trusts the measurements more;
		\item as $R_v \to \infty$, one has
		      \[
			      L_k \to 0,
		      \]
		      so the filter ignores the measurements;
		\item increasing the process noise covariance $Q_w$ increases the predicted covariance, which typically increases the gain and makes the filter trust the measurements more;
		\item decreasing $Q_w$ has the opposite effect and makes the filter trust the model more.
	\end{itemize}
\end{proposition}
\begin{proof}
	The formula for $L_k$ shows immediately that larger $R_v$ enlarges the matrix being inverted in the denominator and therefore suppresses the gain. In the limit
	\[
		R_v \to \infty,
	\]
	the inverse tends to zero, so
	\[
		L_k \to 0.
	\]
	On the other hand, the predicted covariance
	\[
		P_{k|k-1}=A P_{k-1|k-1}A^\top + Q_w
	\]
	increases with $Q_w$. Since $P_{k|k-1}$ appears multiplicatively in the numerator and also enlarges the denominator through
	\[
		CP_{k|k-1}C^\top,
	\]
	the gain increases in the Loewner-order sense along observed directions. Intuitively, larger process noise means the model is less trustworthy, so the filter leans more heavily on measurements.
\end{proof}

\subsection{Steady-State Kalman Filter}

\begin{theorem}[Steady-state convergence conditions]
	Assume
	\[
		R_v \succ 0,
		\qquad
		Q_w \succeq 0,
	\]
	and suppose that
	\begin{itemize}
		\item $(A,C)$ is detectable,
		\item $(A,Q_w^{1/2})$ is stabilizable.
	\end{itemize}
	Then the prior covariance recursion converges to a steady-state value $\bar P \succeq 0$ satisfying the estimator DARE
	\[
		\bar P
		=
		A \bar P A^\top + Q_w
		-
		A \bar P C^\top\left(C\bar P C^\top + R_v\right)^{-1}C\bar P A^\top.
	\]
\end{theorem}
\begin{proof}[Proof sketch]
	The Kalman covariance recursion is monotone on the cone of positive semidefinite matrices and remains bounded under the detectability and stabilizability assumptions. Bounded monotone sequences converge, and continuity of the Riccati map shows that the limit must solve the fixed-point equation. The same hypotheses also guarantee that the associated steady-state observer is stable in the estimation-error dynamics.
\end{proof}

\begin{definition}[Steady-state Kalman filter]
	If
	\[
		P_{k|k-1} \to \bar P,
	\]
	then the corresponding steady-state gain is
	\[
		\bar L = \bar P C^\top(C\bar P C^\top + R_v)^{-1}.
	\]
	The steady-state filter is
	\[
		\hat x_{k|k} = \hat x_{k|k-1} + \bar L(y_k-C\hat x_{k|k-1}),
	\]
	\[
		\hat x_{k+1|k} = A\hat x_{k|k}+Bu_k.
	\]
\end{definition}

\begin{remark}[DARE in disguise]
	The steady-state covariance equation is exactly a DARE, but with the LQR data transposed:
	\[
		A \leftrightarrow A^\top,
		\qquad
		B \leftrightarrow C^\top,
		\qquad
		Q \leftrightarrow Q_w,
		\qquad
		R \leftrightarrow R_v.
	\]
\end{remark}

\begin{remark}[Observability and what fails without it]
	Observability, or more precisely detectability, ensures that unstable modes leave a signature in the measurements. If $(A,C)$ is not detectable, there can exist unstable state components that never affect the outputs. No estimator can recover those components from data, so the estimation error along those directions cannot be driven to zero, and the covariance recursion may fail to converge to a stabilizing steady state.
\end{remark}

\subsection{Duality Between LQR and Kalman Filter}

\begin{theorem}[LQR-Kalman duality]
	The Riccati equation of the Kalman filter is the transpose-dual of the Riccati equation of LQR. Under the mapping
	\[
		A_{\text{LQR}} \leftrightarrow A_{\text{KF}}^\top,
		\qquad
		B_{\text{LQR}} \leftrightarrow C_{\text{KF}}^\top,
		\qquad
		Q_{\text{LQR}} \leftrightarrow Q_w,
		\qquad
		R_{\text{LQR}} \leftrightarrow R_v,
	\]
	the LQR Riccati recursion becomes the Kalman prior-covariance recursion.
\end{theorem}
\begin{proof}
	The LQR Riccati recursion is
	\[
		P_k
		=
		Q + A^\top P_{k+1}A
		-
		A^\top P_{k+1}B(R+B^\top P_{k+1}B)^{-1}B^\top P_{k+1}A.
	\]
	Replace
	\[
		A \mapsto A^\top,
		\qquad
		B \mapsto C^\top,
		\qquad
		Q \mapsto Q_w,
		\qquad
		R \mapsto R_v.
	\]
	Then the recursion becomes
	\[
		\Pi_k
		=
		Q_w + A\Pi_{k+1}A^\top
		-
		A\Pi_{k+1}C^\top(R_v+C\Pi_{k+1}C^\top)^{-1}C\Pi_{k+1}A^\top,
	\]
	which is exactly the Kalman prior-covariance recursion after a shift of the time index:
	\[
		P_{k+1|k}
		=
		A P_{k|k-1}A^\top + Q_w
		-
		A P_{k|k-1}C^\top(CP_{k|k-1}C^\top+R_v)^{-1}CP_{k|k-1}A^\top.
	\]
	So the two Riccati theories are the same algebra viewed through transposed roles of control and sensing.
\end{proof}

\begin{remark}[Meaning of the duality]
	The duality means that:
	\begin{itemize}
		\item stabilizability for control becomes detectability for estimation,
		\item control gain design becomes observer gain design,
		\item the same numerical Riccati solvers apply to both problems after transposition and relabeling.
	\end{itemize}
	In that precise sense, solving one problem immediately solves the other.
\end{remark}

\subsection{Innovation Sequence}

\begin{definition}[Innovation]
	The \emph{innovation} at time $k$ is
	\[
		\nu_k = y_k - C\hat x_{k|k-1}.
	\]
	It is the part of the measurement that was not predictable from past data.
\end{definition}

\begin{proposition}[Innovations are white for a correctly tuned filter]
	Assume the Kalman filter is correctly matched to the system and noise statistics. Then the innovation sequence $\{\nu_k\}$ is zero mean and white:
	\[
		\mathbb{E}[\nu_k]=0,
		\qquad
		\mathbb{E}[\nu_k \nu_j^\top]=0 \quad \text{for } j<k.
	\]
\end{proposition}
\begin{proof}
	By definition,
	\[
		\nu_k = y_k - \mathbb{E}[y_k \mid \mathcal{Y}_{k-1}],
	\]
	because
	\[
		C\hat x_{k|k-1}=C\mathbb{E}[x_k \mid \mathcal{Y}_{k-1}]
		=
		\mathbb{E}[Cx_k \mid \mathcal{Y}_{k-1}]
	\]
	and $\mathbb{E}[v_k \mid \mathcal{Y}_{k-1}] = 0$. Thus $\nu_k$ is a martingale difference:
	\[
		\mathbb{E}[\nu_k \mid \mathcal{Y}_{k-1}] = 0.
	\]
	Taking expectations yields
	\[
		\mathbb{E}[\nu_k]=0.
	\]
	Now let $j<k$. Since $\nu_j$ is measurable with respect to $\mathcal{Y}_{k-1}$,
	\begin{align*}
		\mathbb{E}[\nu_k \nu_j^\top]
		&=
		\mathbb{E}\bigl[\mathbb{E}[\nu_k \nu_j^\top \mid \mathcal{Y}_{k-1}]\bigr] \\
		&=
		\mathbb{E}\bigl[\mathbb{E}[\nu_k \mid \mathcal{Y}_{k-1}] \nu_j^\top\bigr] \\
		&= 0.
	\end{align*}
	So the innovation sequence is white.
\end{proof}

\begin{remark}[Why innovations matter]
	If the innovations cease to look white in practice, then either the model, the assumed noise covariances, or the numerical implementation is wrong. Innovation whiteness is therefore a standard diagnostic for filter health monitoring and covariance tuning.
\end{remark}

\subsection{Extended Kalman Filter}

\begin{definition}[Extended Kalman filter]
	For the nonlinear system
	\begin{align*}
		x_{k+1} &= f(x_k,u_k) + w_k, \\
		y_k &= h(x_k) + v_k,
	\end{align*}
	the extended Kalman filter (EKF) linearizes the dynamics and measurement maps around the current estimate:
	\[
		A_k = \left.\frac{\partial f}{\partial x}\right|_{(\hat x_{k|k},u_k)},
		\qquad
		C_k = \left.\frac{\partial h}{\partial x}\right|_{\hat x_{k|k-1}}.
	\]
	It then runs the standard Kalman recursion using these time-varying Jacobians.
\end{definition}

\begin{remark}[Limitations of the EKF]
	The EKF is only a local linearization method.
	\begin{itemize}
		\item It is no longer globally optimal, even under Gaussian noise.
		\item Large nonlinearities can make the covariance inaccurate.
		\item Poor linearization points can cause divergence.
	\end{itemize}
	So the EKF should be understood as a practical approximation, not as an exact nonlinear filtering theory.
\end{remark}

\begin{remark}[Information filter]
	The information filter works with the inverse covariance
	\[
		Y_k = P_{k|k}^{-1}
	\]
	and the information state
	\[
		\eta_k = Y_k \hat x_{k|k}.
	\]
	In this form, measurement updates become additive in information space:
	\[
		Y_{k|k}=Y_{k|k-1}+C^\top R_v^{-1}C,
		\qquad
		\eta_{k|k}=\eta_{k|k-1}+C^\top R_v^{-1}y_k.
	\]
	This representation is advantageous when measurements are sparse, when many sensors contribute additively, or when distributed sensor fusion is required.
\end{remark}

\subsection{Scalar Kalman Filter Example}

\begin{eg}[Two Kalman-filter iterations by hand]
	Consider the scalar model
	\[
		x_{k+1}=x_k+w_k,
		\qquad
		y_k=x_k+v_k,
	\]
	with
	\[
		w_k \sim \mathcal{N}(0,1),
		\qquad
		v_k \sim \mathcal{N}(0,4).
	\]
	Start with
	\[
		\hat x_{0|-1}=0,
		\qquad
		P_{0|-1}=10,
	\]
	and measurements
	\[
		y_0=3,
		\qquad
		y_1=4.
	\]

	At time $k=0$, the Kalman gain is
	\[
		L_0 = \frac{10}{10+4}=\frac{5}{7}.
	\]
	The filtered estimate is
	\[
		\hat x_{0|0}
		=
		0+\frac{5}{7}(3-0)
		=
		\frac{15}{7}.
	\]
	The filtered covariance is
	\[
		P_{0|0}
		=
		\left(1-\frac{5}{7}\right)10
		=
		\frac{20}{7}.
	\]

	The one-step prediction to time $k=1$ is
	\[
		\hat x_{1|0}=\hat x_{0|0}=\frac{15}{7},
	\]
	\[
		P_{1|0}=P_{0|0}+1=\frac{20}{7}+1=\frac{27}{7}.
	\]

	At time $k=1$, the Kalman gain is
	\[
		L_1
		=
		\frac{27/7}{27/7+4}
		=
		\frac{27}{55}.
	\]
	The innovation is
	\[
		\nu_1 = 4-\frac{15}{7}=\frac{13}{7}.
	\]
	So the updated estimate is
	\[
		\hat x_{1|1}
		=
		\frac{15}{7}+\frac{27}{55}\frac{13}{7}
		=
		\frac{168}{55}.
	\]
	The updated covariance is
	\[
		P_{1|1}
		=
		\left(1-\frac{27}{55}\right)\frac{27}{7}
		=
		\frac{108}{55}.
	\]

	Thus after two iterations,
	\[
		L_0=\frac{5}{7},
		\qquad
