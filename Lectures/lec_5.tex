\lecture{5}{}{Numerical Methods for Trajectory Optimization}

\begin{intuition}
	Lecture 4 supplied the theory: HJB gives a PDE, and PMP gives a coupled two-point boundary value problem. Those descriptions are conceptually complete but rarely computationally friendly. Real trajectory optimization therefore proceeds by \emph{numerical transcriptions}: either solve the PMP boundary value problem indirectly, or discretize the trajectories and solve a finite-dimensional nonlinear program directly. This lecture explains the main families of methods used in practice.
\end{intuition}

\subsection{The Two-Point Boundary Value Problem from PMP}

\begin{remark}
	For a fixed-final-time problem, Pontryagin's principle produces the coupled system
	\begin{align*}
		\dot x(t) & = \partial_p H(t,x(t),u(t),p(t)), \\
		\dot p(t) & = -\partial_x H(t,x(t),u(t),p(t)), \\
		u(t) & \in \argmin_{v \in \mathcal{U}} H(t,x(t),v,p(t)).
	\end{align*}
	The state has an initial condition
	\[
		x(t_0)=x_0,
	\]
	while the costate typically has a terminal condition
	\[
		p(t_f)=\nabla_x \phi(x(t_f))
	\]
	or some transversality variant. This is a \emph{two-point boundary value problem} (TPBVP): part of the boundary data is specified at $t_0$, part at $t_f$.
\end{remark}

\begin{definition}[TPBVP structure]
	A generic TPBVP arising from optimal control can be written abstractly as
	\begin{align*}
		\dot z(t) & = F(t,z(t)), \\
		\Psi(z(t_0),z(t_f)) & = 0,
	\end{align*}
	where
	\[
		z(t)=\begin{bmatrix}x(t) \\ p(t)\end{bmatrix}.
	\]
	The difficulty is that the initial state is known, but the initial costate is usually not.
\end{definition}

\begin{remark}[Why TPBVPs are difficult]
	The difficulty comes from three simultaneous sources:
	\begin{itemize}
		\item the boundary conditions are split between the beginning and end of the horizon,
		\item the state and costate dynamics are coupled through the control law,
		\item the resulting system is frequently nonlinear and sensitive to the unknown costate initialization.
	\end{itemize}
	That sensitivity is exactly why indirect shooting can be powerful on easy problems and fragile on hard ones.
\end{remark}

\subsection{Indirect Shooting Methods}

\begin{intuition}
	Indirect shooting converts the TPBVP into root-finding. One guesses the missing boundary data, integrates the ODEs forward, measures the terminal mismatch, and iterates on the guess until the mismatch vanishes.
\end{intuition}

\begin{definition}[Single shooting for a TPBVP]
	Suppose $x(t_0)=x_0$ is known but $p(t_0)$ is unknown. Let the unknown initial costate be denoted by
	\[
		s=p(t_0).
	\]
	Given a guess $s$, solve the initial value problem
	\begin{align*}
		\dot x(t) & = \partial_p H(t,x(t),u(t),p(t)), \\
		\dot p(t) & = -\partial_x H(t,x(t),u(t),p(t)), \\
		u(t) & \in \argmin_{v \in \mathcal{U}}H(t,x(t),v,p(t)),
	\end{align*}
	with initial conditions
	\[
		x(t_0)=x_0,
		\qquad
		p(t_0)=s.
	\]
	The terminal mismatch defines the \emph{shooting residual}
	\[
		R(s)=\Psi(z_s(t_0),z_s(t_f)).
	\]
	The TPBVP is solved when
	\[
		R(s)=0.
	\]
\end{definition}

\begin{algorithm}[H]
	\DontPrintSemicolon
	\caption{Indirect Single Shooting}
	\KwIn{Initial state $x_0$, horizon $[t_0,t_f]$, initial guess $s^{(0)}$ for the unknown costate, tolerance $\varepsilon$}
	\For{$r=0,1,2,\dots$}{
		Set $p(t_0)=s^{(r)}$ and solve the PMP state-costate ODEs forward in time\;
		Evaluate the terminal residual $R(s^{(r)})$\;
		\If{$\norm{R(s^{(r)})}\le \varepsilon$}{
			return the associated trajectory\;
		}
		Update the guess using Newton, Gauss--Newton, or a quasi-Newton step on $R(s)=0$\;
	}
\end{algorithm}

\begin{remark}
	Single shooting reduces a boundary value problem to a nonlinear system in the unknown initial costate (and possibly final time). The price of this reduction is sensitivity: a small perturbation in the initial guess can cause a large perturbation in the terminal state for unstable or long-horizon systems.
\end{remark}

\begin{eg}[Indirect shooting for minimum-energy control of a double integrator]
	Consider the fixed-time problem
	\begin{align*}
		\min_{u(\cdot)} \quad & \frac{1}{2}\int_0^T u(t)^2\,dt \\
		\text{subject to}\quad & \dot x_1=x_2, \\
		                       & \dot x_2=u, \\
		                       & x(0)=\begin{bmatrix}x_{10} \\ x_{20}\end{bmatrix},
		\qquad
		x(T)=\begin{bmatrix}0 \\ 0\end{bmatrix}.
	\end{align*}
	The Hamiltonian is
	\[
		H=\frac{1}{2}u^2+p_1x_2+p_2u.
	\]
	PMP gives
	\[
		u^\star=-p_2,
		\qquad
		\dot p_1=0,
		\qquad
		\dot p_2=-p_1.
	\]
	Write
	\[
		p_1(t)=a,
		\qquad
		p_2(t)=-at+b,
	\]
	where $(a,b)$ are the unknown initial costate parameters. Then
	\[
		u^\star(t)=at-b.
	\]
	Integrating the state equations yields
	\[
		x_2(t)=x_{20}+\frac{a}{2}t^2-bt,
	\]
	\[
		x_1(t)=x_{10}+x_{20}t+\frac{a}{6}t^3-\frac{b}{2}t^2.
	\]
	The terminal conditions $x_1(T)=0$ and $x_2(T)=0$ define the shooting equations
	\[
		R(a,b)=
		\begin{bmatrix}
			x_{10}+x_{20}T+\dfrac{a}{6}T^3-\dfrac{b}{2}T^2 \\
			x_{20}+\dfrac{a}{2}T^2-bT
		\end{bmatrix}
		=
		0.
	\]
	This is a $2\times 2$ nonlinear root-finding problem, which in this case is actually linear in $(a,b)$. The point of the example is structural: indirect shooting converts the optimal-control boundary conditions into a residual map in the unknown initial costate.
\end{eg}

\subsection{Direct Single Shooting}

\begin{intuition}
	Direct methods optimize the controls themselves rather than solving the PMP boundary value problem. In direct single shooting, the controls are parameterized and the dynamics are used to eliminate the states entirely by forward simulation.
\end{intuition}

\begin{definition}[Direct single shooting]
	Choose a time grid
	\[
		t_0<t_1<\cdots<t_N
	\]
	and parameterize the controls by a finite-dimensional vector
	\[
		w=(w_0,\dots,w_{N-1}),
	\]
	for example by taking piecewise-constant controls
	\[
		u(t)=w_k
		\qquad \text{for } t \in [t_k,t_{k+1}).
	\]
	Using a numerical integrator, define the discrete propagation
	\[
		x_{k+1}=F_k(x_k,w_k),
		\qquad
		x_0 \text{ given}.
	\]
	The OCP is thereby transcribed to the finite-dimensional problem
	\[
		\min_w \; J_{\text{disc}}(w),
	\]
	where the states are implicit functions of the control parameters.
\end{definition}

\begin{algorithm}[H]
	\DontPrintSemicolon
	\caption{Direct Single Shooting}
	\KwIn{Initial state $x_0$, control parameterization $w$, integrator, discretized objective, NLP tolerance}
	\For{outer NLP iterations}{
		Given $w$, simulate the dynamics forward to obtain $x_1,\dots,x_N$\;
		Evaluate the discretized objective and any path/terminal constraint residuals\;
		Compute derivatives by sensitivities, adjoints, or automatic differentiation\;
		Update the control parameter vector using an NLP step\;
	}
\end{algorithm}

\begin{remark}[Advantages and disadvantages]
	Direct single shooting has several attractive features:
	\begin{itemize}
		\item very few decision variables if only controls are optimized,
		\item exact satisfaction of the discretized dynamics by construction,
		\item easy implementation when a simulator already exists.
	\end{itemize}
	It also has serious limitations:
	\begin{itemize}
		\item the map from controls to terminal state can be badly conditioned,
		\item state constraints are awkward because states are not decision variables,
		\item numerical integration errors are buried inside the objective and constraints.
	\end{itemize}
\end{remark}

\subsection{Direct Multiple Shooting}

\begin{intuition}
	Multiple shooting keeps the simulation-based flavor of shooting but introduces intermediate state variables. Instead of forcing one long simulation to carry all information, it breaks the trajectory into shorter segments and enforces continuity between them.
\end{intuition}

\begin{definition}[Multiple shooting transcription]
	Given a grid $t_0<\cdots<t_N$, introduce decision variables
	\[
		x_0,\dots,x_N,
		\qquad
		u_0,\dots,u_{N-1}.
	\]
	On each interval $[t_k,t_{k+1}]$, integrate the dynamics with initial value $x_k$ and control parameter $u_k$ to produce the segment endpoint
	\[
		\Phi_k(x_k,u_k).
	\]
	The transcription enforces the continuity constraints
	\[
		x_{k+1}-\Phi_k(x_k,u_k)=0,
		\qquad
		k=0,\dots,N-1.
	\]
\end{definition}

\begin{algorithm}[H]
	\DontPrintSemicolon
	\caption{Direct Multiple Shooting}
	\KwIn{Initial guess for node states and controls, segment integrator, discretized objective}
	\For{outer NLP iterations}{
		For each interval, integrate from $x_k$ under $u_k$ to compute $\Phi_k(x_k,u_k)$\;
		Assemble continuity defects $x_{k+1}-\Phi_k(x_k,u_k)$\;
		Assemble the NLP objective and all constraints\;
		Solve the structured NLP step and update node states and controls\;
	}
\end{algorithm}

\begin{remark}
	Multiple shooting is often markedly better conditioned than single shooting because each integration window is short. The state variables at grid points also make path and boundary constraints much easier to impose directly.
\end{remark}

\begin{proposition}[Sparsity structure of multiple shooting]
	The Jacobian of the multiple-shooting continuity constraints is block sparse. Specifically, each constraint
	\[
		c_k(x_k,x_{k+1},u_k)=x_{k+1}-\Phi_k(x_k,u_k)
	\]
	depends only on the triplet $(x_k,u_k,x_{k+1})$.
\end{proposition}
\begin{proof}
	By construction, the $k$th continuity equation compares the next node state $x_{k+1}$ with the endpoint of the segment launched from $x_k$ under $u_k$. No other node states or controls enter this equation. Therefore the derivative of $c_k$ with respect to any variable outside $(x_k,u_k,x_{k+1})$ is zero. Stacking the constraints over all intervals produces a block-banded Jacobian.
\end{proof}

\begin{remark}
	This block sparsity is one of the reasons multiple shooting is a standard transcription in software such as CasADi- and ACADOS-based pipelines. Sparse linear algebra is not a minor implementation detail here; it is what makes large problems tractable.
\end{remark}

\subsection{Direct Collocation}

\begin{intuition}
	Collocation abandons explicit forward simulation and instead approximates the state trajectory by low-order polynomials on each interval. The dynamics are enforced at selected collocation points, turning differential equations into algebraic equality constraints.
\end{intuition}

\begin{definition}[Trapezoidal collocation]
	Let the step size be
	\[
		h_k=t_{k+1}-t_k.
	\]
	With node variables $(x_k,u_k)$ and $(x_{k+1},u_{k+1})$, trapezoidal collocation enforces
	\[
		x_{k+1}-x_k-\frac{h_k}{2}\bigl(f(x_k,u_k)+f(x_{k+1},u_{k+1})\bigr)=0.
	\]
\end{definition}

\begin{definition}[Hermite--Simpson collocation]
	Hermite--Simpson collocation introduces a midpoint state and control:
	\[
		u_{k+\frac{1}{2}}=\frac{u_k+u_{k+1}}{2},
	\]
	\[
		x_{k+\frac{1}{2}}
		=
		\frac{x_k+x_{k+1}}{2}
		+
		\frac{h_k}{8}\bigl(f(x_k,u_k)-f(x_{k+1},u_{k+1})\bigr).
	\]
	The defect constraint is
	\[
		x_{k+1}-x_k
		-
		\frac{h_k}{6}
		\Bigl(
			f(x_k,u_k)+4f(x_{k+\frac{1}{2}},u_{k+\frac{1}{2}})+f(x_{k+1},u_{k+1})
		\Bigr)
		=0.
	\]
\end{definition}

\begin{algorithm}[H]
	\DontPrintSemicolon
	\caption{Direct Collocation Transcription}
	\KwIn{Time grid, initial guess for node states and controls, collocation rule}
	Choose optimization variables containing all node states and controls\;
	For each interval, form the collocation defect equations\;
	Approximate the continuous cost by a quadrature rule consistent with the transcription\;
	Add boundary, path, and control constraints at nodes (and midpoints if needed)\;
	Solve the resulting sparse NLP with SQP or an interior-point method\;
\end{algorithm}

\begin{remark}
	In direct collocation, both states and controls are explicit decision variables. This is the opposite design choice from single shooting. The reward is flexibility: state constraints, path constraints, and stiff dynamics are often handled more robustly.
\end{remark}

\begin{proposition}[Sparsity of direct collocation NLPs]
	Direct collocation produces a large but sparse nonlinear program. Each defect constraint couples only neighboring nodes and, in midpoint formulas, a small number of auxiliary midpoint variables.
\end{proposition}
\begin{proof}
	The collocation equation on interval $k$ uses only the variables attached to that interval, such as
	\[
		(x_k,u_k,x_{k+1},u_{k+1})
	\]
	for trapezoidal collocation, or the same tuple together with midpoint data for Hermite--Simpson. Consequently, the Jacobian row corresponding to interval $k$ has nonzero entries only in a small local block. After stacking all intervals, the overall Jacobian is banded or nearly banded, and the Hessian of the Lagrangian inherits comparable locality.
\end{proof}

\begin{remark}
	Collocation methods often behave better than explicit shooting on stiff systems because the dynamics are enforced implicitly through algebraic constraints rather than through one long explicit integration chain.
\end{remark}

\subsection{Transcription to Nonlinear Programming}

\begin{definition}[General NLP form from direct transcription]
	After discretization, most direct trajectory-optimization methods produce an NLP of the form
	\begin{align*}
		\min_{z} \quad & F(z) \\
		\text{subject to}\quad & c(z)=0, \\
		                       & d(z)\le 0,
	\end{align*}
	where $z$ stacks the discretized state, control, and possibly time variables.
\end{definition}

\begin{remark}[Connection to Lecture 3]
	Once transcribed, the problem is a standard finite-dimensional constrained optimization problem. The KKT conditions, Lagrange multipliers, SQP subproblems, interior-point barrier terms, and dual variables are exactly those of Lecture 3. Dynamic optimal control becomes large-scale nonlinear programming after transcription.
\end{remark}

\begin{remark}[Sparsity and band structure]
	If the grid is ordered in time, then most constraints only couple nearby time indices. This yields banded Jacobians and structured Hessians of the Lagrangian. Efficient solvers exploit this structure through sparse factorizations, condensing, partial condensing, or Riccati-type eliminations.
\end{remark}

\begin{remark}[Warm starts and real-time use]
	Optimal control is often solved repeatedly with slowly changing initial conditions or references. Warm-starting from the previous solution can dramatically reduce solve time. This is one of the key reasons direct methods become viable in real-time MPC and nonlinear receding-horizon control.
\end{remark}

\subsection{Comparison of Major Trajectory-Optimization Methods}

\begin{center}
	\small
	\begin{adjustbox}{max width=\textwidth}
	\begin{tabular}{p{0.18\textwidth}p{0.22\textwidth}p{0.22\textwidth}p{0.28\textwidth}}
		\toprule
		Method & Decision variables & Strengths & Weaknesses \\
		\midrule
		Indirect shooting & Initial costate (and possibly final time) & Uses exact PMP structure; low dimension when it works & Sensitive, nonconvex, hard to initialize, awkward with constraints \\
		Direct single shooting & Control parameters only & Simple to implement; few variables; reuses simulators & Poor conditioning; state constraints awkward; long-horizon sensitivity \\
		Direct multiple shooting & Node states and controls & Better conditioning; path constraints natural; sparse NLP & More variables than single shooting; requires continuity constraints \\
		Direct collocation & Node states and controls (plus possible midpoint data) & Very sparse; handles path constraints and stiff dynamics well & Larger NLP; transcription design affects accuracy and conditioning \\
		\bottomrule
	\end{tabular}
	\end{adjustbox}
\end{center}

\begin{remark}
	In modern robotics practice, direct multiple shooting and direct collocation dominate general-purpose nonlinear trajectory optimization. Indirect methods remain valuable when an accurate analytical model is available and the structure is favorable, but they are usually less robust in the presence of path constraints, hybrid effects, or weak initial guesses.
\end{remark}
